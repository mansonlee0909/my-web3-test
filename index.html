// 1. 確認地址（這是你截圖左下角的地址） const contractAddress = "0xDe467Cae710d0bFF7a9df6eBB4bA8D51Dbc1BbB5"; // 2. 簡化版 ABI，減少解碼錯誤 const abi = [ "function data() public view returns (uint256)", "function set(uint256 _data) public" ]; let provider, contract; async function init() { if (window.ethereum) { try { // 強制請求連線 await window.ethereum.request({ method: 'eth_requestAccounts' }); // 使用 BrowserProvider provider = new ethers.BrowserProvider(window.ethereum); // 【重要：檢查網路】 const network = await provider.getNetwork(); console.log("當前連線網路 ID:", network.chainId); // 如果不是 Sepolia (ID: 11155111)，強制報錯提醒 if (network.chainId !== 11155111n) { document.getElementById('status').innerText = "錯誤：請將 MetaMask 切換到 Sepolia 測試網！"; document.getElementById('displayData').innerText = "網路錯誤"; return; } contract = new ethers.Contract(contractAddress, abi, provider); // 讀取數據 const val = await contract.data(); document.getElementById('displayData').innerText = val.toString(); document.getElementById('status').innerText = "連線成功 (Sepolia)"; } catch (err) { console.error("詳細錯誤資訊:", err); document.getElementById('status').innerText = "讀取失敗，請確認合約已部署"; } } else { document.getElementById('status').innerText = "未安裝 MetaMask"; } }
